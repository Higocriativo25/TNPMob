function PRIMEIRAPAG_Login_func_logar() { const PRIMEIRAPAG_email = document.getElementById('PRIMEIRAPAG_Login_input_email').value.trim(); const PRIMEIRAPAG_pass = document.getElementById('PRIMEIRAPAG_Login_input_password').value; const PRIMEIRAPAG_tipo = document.getElementById('PRIMEIRAPAG_Login_select_tipo_login').value; if (!PRIMEIRAPAG_email || !PRIMEIRAPAG_pass) { return PRIMEIRAPAG_Login_func_mostrarErro( 'PRIMEIRAPAG_Login_error_main', 'Preencha e-mail e senha para entrar.' ); } if (!PRIMEIRAPAG_Login_func_isEmailValid(PRIMEIRAPAG_email)) { return PRIMEIRAPAG_Login_func_mostrarErro( 'PRIMEIRAPAG_Login_error_main', 'Digite um e-mail válido.' ); } PRIMEIRAPAG_Login_auth.signInWithEmailAndPassword(PRIMEIRAPAG_email, PRIMEIRAPAG_pass) .then(async (userCredential) => { const PRIMEIRAPAG_userAuth = userCredential.user; const PRIMEIRAPAG_uid = PRIMEIRAPAG_userAuth.uid; const PRIMEIRAPAG_docUsuario = await PRIMEIRAPAG_Login_db .collection('usuarios') .doc(PRIMEIRAPAG_uid) .get(); if (!PRIMEIRAPAG_docUsuario.exists) { if (!PRIMEIRAPAG_userAuth.emailVerified) { const emailDestino = document.getElementById('SEGUNDAPAG_Login_email_destino'); if (emailDestino) { emailDestino.innerText = PRIMEIRAPAG_userAuth.email; } PRIMEIRAPAG_Login_func_irPara('SEGUNDAPAG_Login_screen_confirmar_email'); return; } PRIMEIRAPAG_Login_func_irPara('SEGUNDAPAG_Login_screen_cadastro_rapido'); return; } const PRIMEIRAPAG_user = PRIMEIRAPAG_docUsuario.data(); // LOGIN COMO PASSAGEIRO if (PRIMEIRAPAG_tipo === 'passageiro') { PRIMEIRAPAG_Login_func_retornarEtapaCadastro(PRIMEIRAPAG_user); return; } // LOGIN COMO MOTORISTA if (PRIMEIRAPAG_tipo === 'motorista') { if ( PRIMEIRAPAG_user.perfis && PRIMEIRAPAG_user.perfis.motorista === true ) { const PRIMEIRAPAG_docMotorista = await PRIMEIRAPAG_Login_db .collection('motoristas') .doc(PRIMEIRAPAG_uid) .get(); if (PRIMEIRAPAG_docMotorista.exists) { const PRIMEIRAPAG_motorista = PRIMEIRAPAG_docMotorista.data(); const PRIMEIRAPAG_etapaUsuario = PRIMEIRAPAG_user.etapaCadastro || ""; const PRIMEIRAPAG_etapaMotorista = PRIMEIRAPAG_motorista.etapaCadastroMotorista || ""; // 1. Se ainda não enviou selfie de perfil, volta para selfie if ( PRIMEIRAPAG_user.selfieEnviada !== true || PRIMEIRAPAG_etapaUsuario === "selfie" ) { PRIMEIRAPAG_Login_func_irPara('QUINTAPAG_Login_screen_selfie'); return; } // 2. Se está na etapa de categoria do motorista if ( PRIMEIRAPAG_etapaUsuario === "motorista_categoria" || PRIMEIRAPAG_etapaMotorista === "categoria" ) { if (document.getElementById('MOTORISTAPAG_Login_screen_categoria')) { PRIMEIRAPAG_Login_func_irPara('MOTORISTAPAG_Login_screen_categoria'); } else { alert("Você está na etapa de categoria do motorista. Agora vamos criar essa tela."); PRIMEIRAPAG_Login_func_irPara('PRIMEIRAPAG_Login_screen_welcome_motorista'); } return; } // 3. Se está na etapa de CNH if ( PRIMEIRAPAG_etapaUsuario === "motorista_cnh" || PRIMEIRAPAG_etapaMotorista === "cnh" ) { if (document.getElementById('MOTORISTAPAG_Login_screen_cnh')) { PRIMEIRAPAG_Login_func_irPara('MOTORISTAPAG_Login_screen_cnh'); } else { alert("Você está na etapa de CNH. Essa tela será criada no próximo passo."); PRIMEIRAPAG_Login_func_irPara('PRIMEIRAPAG_Login_screen_welcome_motorista'); } return; } const PRIMEIRAPAG_statusMotorista = PRIMEIRAPAG_motorista.status ? PRIMEIRAPAG_motorista.status.geral : ""; // 4. Se já estiver aprovado if (PRIMEIRAPAG_statusMotorista === "aprovado") { PRIMEIRAPAG_Login_func_irPara('PRIMEIRAPAG_Login_screen_welcome_motorista'); return; } // 5. Caso esteja incompleto ou em análise PRIMEIRAPAG_Login_func_irPara('PRIMEIRAPAG_Login_screen_welcome_motorista'); return; } } alert("Você ainda não iniciou ou concluiu o cadastro de motorista. Entre como passageiro para continuar."); PRIMEIRAPAG_Login_func_retornarEtapaCadastro(PRIMEIRAPAG_user); return; } }) .catch((erro) => { console.error("ERRO REAL NO LOGIN:", erro.code, erro.message); let mensagem = "Erro ao entrar. Verifique seu e-mail e senha."; if (erro.code === "auth/invalid-credential") { mensagem = "E-mail ou senha incorretos. Confira os dados ou use Esqueci minha senha."; } if (erro.code === "auth/wrong-password") { mensagem = "Senha incorreta. Confira a senha ou use Esqueci minha senha."; } if (erro.code === "auth/user-not-found") { mensagem = "Este e-mail ainda não possui cadastro."; } if (erro.code === "auth/too-many-requests") { mensagem = "Muitas tentativas de login. Aguarde alguns minutos e tente novamente."; } if (erro.code === "auth/user-disabled") { mensagem = "Esta conta está desativada. Fale com o suporte."; } PRIMEIRAPAG_Login_func_mostrarErro( 'PRIMEIRAPAG_Login_error_main', mensagem ); }); }
